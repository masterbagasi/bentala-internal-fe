> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build an AI Studio section (Pencari Ide, Content Builder, BPI Intelligence) powered by Claude and free news sources, integrated into the existing Bentala dashboard.

**Architecture:** Server-side API routes handle all Claude/RSS/Reddit calls. UI components follow existing inline-style pattern. News cached in Supabase `news_cache` table (30-min TTL). Export to BPI/BSI creates a draft `posts` row.

**Tech Stack:** Next.js 14 App Router, @anthropic-ai/sdk, fast-xml-parser, Supabase, React inline styles (no Tailwind custom classes)

```ts
import { XMLParser } from 'fast-xml-parser'
import { NextResponse } from 'next/server'
import { createClient } from '@supabase/supabase-js'

const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!
)

const RSS_SOURCES = [
  { key: 'bbc',              type: 'international', url: 'https://feeds.bbci.co.uk/news/world/rss.xml' },
  { key: 'aljazeera',        type: 'international', url: 'https://www.aljazeera.com/xml/rss/all.xml' },
  { key: 'google_indonesia', type: 'international', url: 'https://news.google.com/rss/search?q=Indonesia&hl=en&gl=US&ceid=US:en' },
  { key: 'kompas',           type: 'indonesia',     url: 'https://rss.kompas.com/internasional' },
  { key: 'detik',            type: 'indonesia',     url: 'https://rss.detik.com/index.php/detikcom' },
  { key: 'tempo',            type: 'indonesia',     url: 'https://rss.tempo.co/nasional' },
  { key: 'cnnindonesia',     type: 'indonesia',     url: 'https://www.cnnindonesia.com/rss' },
]

const REDDIT_SOURCES = [
  { key: 'reddit_indonesia', type: 'social', url: 'https://www.reddit.com/r/indonesia/hot.json?limit=25' },
  { key: 'reddit_worldnews', type: 'social', url: 'https://www.reddit.com/r/worldnews/search.json?q=Indonesia&sort=hot&limit=25' },
]

// eslint-disable-next-line @typescript-eslint/no-explicit-any
async function fetchRSS(src: typeof RSS_SOURCES[number]): Promise<any[]> {
  try {
    const res = await fetch(src.url, { headers: { 'User-Agent': 'BentalaBot/1.0' } })
    if (!res.ok) return []
    const xml = await res.text()
    const parser = new XMLParser()
    const parsed = parser.parse(xml)
    // eslint-disable-next-line @typescript-eslint/no-explicit-any
    const raw: any[] = parsed?.rss?.channel?.item ?? parsed?.feed?.entry ?? []
    const items = Array.isArray(raw) ? raw : [raw]
    return items.slice(0, 10).map(item => ({
      source: src.key,
      source_type: src.type,
      title: String(item.title ?? '').replace(/<[^>]*>/g, '').trim(),
      summary: String(item.description ?? item.summary ?? '').replace(/<[^>]*>/g, '').slice(0, 200).trim(),
      url: String(item.link ?? item.guid ?? '').trim(),
      published_at: (() => {
        try { return new Date(String(item.pubDate ?? item.published ?? '')).toISOString() }
        catch { return new Date().toISOString() }
      })(),
    })).filter(i => i.title && i.url)
  } catch { return [] }
}

// eslint-disable-next-line @typescript-eslint/no-explicit-any
async function fetchReddit(src: typeof REDDIT_SOURCES[number]): Promise<any[]> {
  try {
    const res = await fetch(src.url, { headers: { 'User-Agent': 'BentalaBot/1.0' } })
    if (!res.ok) return []
    const data = await res.json()
    // eslint-disable-next-line @typescript-eslint/no-explicit-any
    const posts: any[] = data?.data?.children ?? []
    return posts.slice(0, 10).map(p => ({
      source: src.key,
      source_type: src.type,
      title: String(p.data.title ?? '').trim(),
      summary: String(p.data.selftext ?? '').slice(0, 200).trim(),
      url: `https://reddit.com${p.data.permalink ?? ''}`,
      published_at: new Date((p.data.created_utc ?? 0) * 1000).toISOString(),
    })).filter(i => i.title && i.url)
  } catch { return [] }
}

export async function GET() {
  try {
    const thirtyMinAgo = new Date(Date.now() - 30 * 60 * 1000).toISOString()
    const { data: cached } = await supabase
      .from('news_cache')
      .select('*')
      .gt('fetched_at', thirtyMinAgo)
      .order('published_at', { ascending: false })
      .limit(100)

    if (cached && cached.length > 0) {
      return NextResponse.json({ items: cached, from_cache: true })
    }

    const [rssAll, redditAll] = await Promise.all([
      Promise.all(RSS_SOURCES.map(fetchRSS)),
      Promise.all(REDDIT_SOURCES.map(fetchReddit)),
    ])

    const allItems = [...rssAll.flat(), ...redditAll.flat()]

    if (allItems.length > 0) {
      await supabase.from('news_cache').upsert(
        allItems.map(item => ({ ...item, fetched_at: new Date().toISOString() })),
        { onConflict: 'url' }
      )
    }

    const { data: fresh } = await supabase
      .from('news_cache')
      .select('*')
      .order('published_at', { ascending: false })
      .limit(100)

    return NextResponse.json({ items: fresh ?? allItems, from_cache: false })
  } catch (err) {
    console.error('[/api/ai/news]', err)
    return NextResponse.json({ error: 'Gagal fetch berita' }, { status: 500 })
  }
}
```
